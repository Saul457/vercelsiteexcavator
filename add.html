async function sendToTelegram() { const photoInput = document.getElementById('machine-image'); const description = document.getElementById('machine-desc').value; const btn = document.getElementById('send-btn'); if (photoInput.files.length === 0 || description.trim() === "") { alert("الرجاء إكمال كافة البيانات أولاً!"); return; } btn.innerText = "⏳ جاري الإرسال..."; btn.disabled = true; const formData = new FormData(); formData.append('photo', photoInput.files[0]); formData.append('caption', description); try { const response = await axios.post('/api/send-telegram', formData, { headers: { 'Content-Type': 'multipart/form-data' } }); if (response.status === 200) { alert("✅ تم الإرسال بنجاح!"); document.getElementById('machine-desc').value = ""; photoInput.value = ""; } } catch (error) { alert("❌ فشل الإرسال."); } finally { btn.innerText = "إرسال الطلب بشكل آمن"; btn.disabled = false; } }
